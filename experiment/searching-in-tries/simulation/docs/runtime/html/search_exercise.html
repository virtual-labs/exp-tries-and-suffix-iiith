1 Features of the artefact

  • This aretefact implements the practice for searching in Tries
  • It prints out a word that the user needs to search for
  • As the user clicks the nodes they get highlighted
  • The user can check his answer, all the correct nodes will be colored gree, red/black otherwise.
  • Next, a new question will start off

2.8.2 Declaration of the variables that initialize the trie

The words that will be present in the trie and the words which will not be, all comprising a list of possible queries is defined here. Then the bindings for the buttons and activation of the nodes to a click response is generated. 

<script type="text/javascript">
let positive = ["Hello", "Friend", "Animesh", "Ant", "Apple", "Asterisk"]
let negative = ["Hibernate", "Hysteria", "Friday", "App"]
var searchExercise = new SearchExercise("#render-2", "#bindings-2", positive, negative);

$('#searchexercise-searching-for').html(searchExercise.selectedWord);

$('#found-searchexercise').click(function () {
    if (searchExercise.check(true) === true) $('#searchexercise-result').html('You are Right.');
    else $('#searchexercise-result').html('Sorry, Please try again.');
});
$('#failed-searchexercise').click(function () {
    if (searchExercise.check(false) === true) $('#searchexercise-result').html('You are Right.');
    else $('#searchexercise-result').html('Sorry, Please try again.');
});
$('#reset-searchexercise').click(function () {
    searchExercise.begin();
    $('#searchexercise-searching-for').html(searchExercise.selectedWord);
    $('#searchexercise-result').html('Let\'s Try searching some words');
});
</script>
